    <script>
        const firebaseConfig = {
            apiKey: "AIzaSyA3E8fbd9-Uw2VAAoTyb1ltLYLDE2PTWTw",
            databaseURL: "https://sensores-temp-y-humedad-default-rtdb.firebaseio.com",
            projectId: "sensores-temp-y-humedad",
        };
        firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        // Escuchar cambios en sensores
        db.ref('sensor').on('value', (snap) => {
            const data = snap.val();
            document.getElementById('temp').innerText = data.temperatura;
            document.getElementById('tierra').innerText = data.tierra;
        });

        // Función para el botón
        void function toggleBomba(estado) {
            db.ref('actuadores').update({ bomba: estado });
        }
    </script>
